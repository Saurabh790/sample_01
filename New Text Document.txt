-- duplicates inside same batch by TCN + line
SELECT HDR_CLM_TCN, LI_NUM, COUNT(*) cnt
FROM CA_UNIVERSE_BATCH_T
WHERE CT_BATCH_ID = 451
GROUP BY HDR_CLM_TCN, LI_NUM
HAVING COUNT(*) > 1;
