SELECT txn_id,
       customer_id,
       product_id,
       txn_amount,
       invoice_date,
       status,
       run_id
FROM TXN_SOURCE
ORDER BY txn_id;


SELECT stage_id,
       run_id,
       txn_id,
       customer_id,
       product_id,
       base_amount,
       discount,
       tax,
       commission,
       net_amount,
       created_at
FROM TXN_STAGE
ORDER BY stage_id;


SELECT txn_id,
       run_id,
       customer_id,
       invoice_date,
       base_amount,
       discount,
       tax,
       commission,
       net_amount,
       created_at
FROM FINAL_TXN_LINE
ORDER BY txn_id;


SELECT error_id,
       run_id,
       txn_id,
       customer_id,
       product_id,
       txn_amount,
       error_reason,
       created_at
FROM TXN_ERROR
ORDER BY error_id;
	
	
	SELECT invoice_id,
       run_id,
       customer_id,
       invoice_date,
       total_gross,
       total_discount,
       total_tax,
       total_commission,
       total_net,
       created_at
FROM INVOICE_FINAL
ORDER BY invoice_id;
